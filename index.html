1. Turn ON LED When Button is Pressed

Arduino UNO

  • LED + → Pin 13
  • Button → Pin 2
  • GND → GND

int buttonPin = 2;
int ledPin = 13;

void setup() {
  pinMode(buttonPin, INPUT_PULLUP);
  pinMode(ledPin, OUTPUT);
}

void loop() {

  if(digitalRead(buttonPin) == LOW) {
    digitalWrite(ledPin, HIGH);
  }
  else {
    digitalWrite(ledPin, LOW);
  }

}

Raspberry Pi

  • LED → GPIO17
  • Button → GPIO27
  • GND → GND
Install: 

sudo apt install python3-rpi.gpio


import RPi.GPIO as GPIO
import time

LED = 17
BUTTON = 27

GPIO.setmode(GPIO.BCM)

GPIO.setup(LED, GPIO.OUT)
GPIO.setup(BUTTON, GPIO.IN,
pull_up_down=GPIO.PUD_UP)

while True:

    if GPIO.input(BUTTON) == 0:
        GPIO.output(LED, 1)
    else:
        GPIO.output(LED, 0)

    time.sleep(0.1)

2. IR Sensor Detection with LED & Buzzer

Arduino UNO

  • IR OUT → Pin 2
  • LED → Pin 8
  • Buzzer → Pin 9

int irSensor = 2;
int led = 8;
int buzzer = 9;

void setup() {

  pinMode(irSensor, INPUT);
  pinMode(led, OUTPUT);
  pinMode(buzzer, OUTPUT);

}

void loop() {

  if(digitalRead(irSensor) == LOW) {

    digitalWrite(led, HIGH);
    digitalWrite(buzzer, HIGH);

  } else {

    digitalWrite(led, LOW);
    digitalWrite(buzzer, LOW);

  }

}

Raspberry Pi

  • IR → GPIO18
  • LED → GPIO23
  • Buzzer → GPIO24

sudo apt install python3-rpi.gpio


import RPi.GPIO as GPIO
import time

IR = 18
LED = 23
BUZZER = 24

GPIO.setmode(GPIO.BCM)

GPIO.setup(IR, GPIO.IN)
GPIO.setup(LED, GPIO.OUT)
GPIO.setup(BUZZER, GPIO.OUT)

while True:

    if GPIO.input(IR) == 0:

        GPIO.output(LED, 1)
        GPIO.output(BUZZER, 1)

    else:

        GPIO.output(LED, 0)
        GPIO.output(BUZZER, 0)

    time.sleep(0.1)

3. LDR Controlled LED

Arduino UNO

  • LDR → A0
  • LED → Pin 13

int ldr = A0;
int led = 13;

void setup() {

  pinMode(led, OUTPUT);

}

void loop() {

  int value = analogRead(ldr);

  if(value < 500) {
    digitalWrite(led, HIGH);
  }
  else {
    digitalWrite(led, LOW);
  }

}

Raspberry Pi

  • LDR → MCP3008 CH0
  • LED → GPIO17

pip3 install gpiozero spidev


from gpiozero import MCP3008, LED
from time import sleep

ldr = MCP3008(channel=0)
led = LED(17)

while True:

    value = ldr.value

    if value < 0.5:
        led.on()
    else:
        led.off()

    sleep(0.1)

4. DHT22 Temperature & Humidity

Arduino UNO

  • DHT22 DATA → Pin 2
  • VCC → 5V
  • GND → GND

#include 

#define DHTPIN 2
#define DHTTYPE DHT22

DHT dht(DHTPIN, DHTTYPE);

void setup() {

  Serial.begin(9600);
  dht.begin();

}

void loop() {

  float temp =
  dht.readTemperature();

  float hum =
  dht.readHumidity();

  Serial.println(temp);
  Serial.println(hum);

  delay(2000);

}

Raspberry Pi

  • DHT22 DATA → GPIO4
  • VCC → 3.3V
  • GND → GND

pip3 install Adafruit_DHT

    Adafruit_DHT.read_retry(
    sensor, pin)

    print(humidity)

    time.sleep(2)

5. Raspberry Pi TCP Server & Client

TCP Server

  • Runs on Raspberry Pi
  • Sends a text message to client
  • Uses Python socket library

import socket

HOST = '0.0.0.0'
PORT = 5000

server = socket.socket(
socket.AF_INET,
socket.SOCK_STREAM
)

server.bind((HOST, PORT))

server.listen(1)

print("Waiting for client...")

client, address = server.accept()

print("Connected to:", address)

message = "Hello from Raspberry Pi Server!"

client.send(message.encode())

client.close()
server.close()

TCP Client

  • Receives message from server
  • Works on another Raspberry Pi or PC
  • Replace IP with server IP

import socket

SERVER_IP = '192.168.1.10'
PORT = 5000

client = socket.socket(
socket.AF_INET,
socket.SOCK_STREAM
)

client.connect((SERVER_IP, PORT))

message = client.recv(1024)

print(
"Message from server:",
message.decode()
)

client.close()

Run Server


python3 server.py

Run Client


python3 client.py

6. DHT11 Data Upload to ThingSpeak

Project Description

  • Reads temperature & humidity using DHT11
  • Sends sensor data to ThingSpeak Cloud
  • Uses Raspberry Pi + Python
  • Uploads data every 15 seconds

Pin Connections

  • DHT11 DATA → GPIO4
  • VCC → 3.3V
  • GND → GND

Install Dependencies


sudo apt update

sudo apt install python3-pip

pip3 install Adafruit_DHT

pip3 install requests

Python Code 


import Adafruit_DHT
import requests
import time

API_KEY = 'YOUR_API_KEY'

BASE_URL =
'https://api.thingspeak.com/update'

sensor = Adafruit_DHT.DHT11

pin = 4

while True:

    humidity, temperature =
    Adafruit_DHT.read(sensor, pin)

    if humidity is not None and \
       temperature is not None:

        payload = {

            'api_key': API_KEY,
            'field1': temperature,
            'field2': humidity

        }

        response = requests.get(
        BASE_URL,
        params=payload
        )

        print("Data Sent")
        print(response.text)

    time.sleep(15)

Run Code


python3 thingspeak.py
